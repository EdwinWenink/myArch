# Add new user. -m flag creates home directory. -s passes shell
- useradd -m -g users -G wheel -s /bin/bash username # perhaps not use -G wheel
- passwd

# Enable sudo use for wheel members
- visudo uncomment wheel ALL=(ALL) ALL
